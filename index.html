  from browser import document
    
    import numpy as np

    def calculate(list):
      if len(list) < 9:
        raise ValueError("List must contain nine numbers.")

      # Convert list to numpy array and reshape to 3x3  
      array = np.array(list).reshape(3, 3)

      # Calculate statistics
      mean = [array.mean(axis=0).tolist(), array.mean(axis=1).tolist(), array.mean()]
      variance = [array.var(axis=0).tolist(), array.var(axis=1).tolist(), array.var()]
      std_dev = [array.std(axis=0).tolist(), array.std(axis=1).tolist(), array.std()]
      maxi = [array.max(axis=0).tolist(), array.max(axis=1).tolist(), array.max()]
      mini = [array.min(axis=0).tolist(), array.min(axis=1).tolist(), array.min()]
      summation = [array.sum(axis=0).tolist(), array.sum(axis=1).tolist(), array.sum()]

      # Return results in a dictionary
      return {
              'mean': mean,
              'variance': variance,
              'standard deviation': std_dev,
              'max': maxi,
              'min': mini,
              'sum': summation
             }

      # código para obtener números, llamar calculate() y mostrar resultado
